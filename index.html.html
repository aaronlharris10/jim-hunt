// ============================================================ // JIM HUNT — Google Apps Script Backend // Paste this entire file into your Apps Script editor // ============================================================ function doGet(e) { var action = e.parameter.action; if (action === "getEntries") { return getEntries(); } return ContentService.createTextOutput(JSON.stringify({ status: "ok" })) .setMimeType(ContentService.MimeType.JSON); } function doPost(e) { var data = JSON.parse(e.postData.contents); if (data.action === "addEntry") { return addEntry(data.name, data.guess); } return ContentService.createTextOutput(JSON.stringify({ status: "error" })) .setMimeType(ContentService.MimeType.JSON); } function getEntries() { var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Entries"); var rows = sheet.getDataRange().getValues(); var entries = []; for (var i = 1; i < rows.length; i++) { // skip header row if (rows[i][0]) { entries.push({ name: rows[i][0], guess: rows[i][1] }); } } var result = JSON.stringify({ entries: entries }); return ContentService.createTextOutput(result) .setMimeType(ContentService.MimeType.JSON); } function addEntry(name, guess) { var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Entries"); var rows = sheet.getDataRange().getValues(); // Check for duplicate name (case-insensitive) for (var i = 1; i < rows.length; i++) { if (rows[i][0] && rows[i][0].toString().toLowerCase() === name.toLowerCase()) { return ContentService.createTextOutput(JSON.stringify({ status: "duplicate" })) .setMimeType(ContentService.MimeType.JSON); } } // Append new entry with timestamp sheet.appendRow([name, guess, new Date()]); return ContentService.createTextOutput(JSON.stringify({ status: "success" })) .setMimeType(ContentService.MimeType.JSON); }
